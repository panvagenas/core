fields[file] => (string)name fields[___file_info][file] = array('name' => '', 'tmp_name' => '', etc.)

files[0] => (string)name files[___file_info][0] = array('name' => '', 'tmp_name' => '', etc.)

files[1] => (string)name files[___file_info][1] = array('name' => '', 'tmp_name' => '', etc.)

Example #4: <input type="file" name="fields[files][]" multiple="multiple" />

fields[files][0] => (string)name fields[files][___file_info][0] = array('name' => '', 'tmp_name' => '', etc.)
